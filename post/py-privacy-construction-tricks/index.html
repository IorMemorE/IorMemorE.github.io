# privacy_construct_example.py
class __Token: ...

_token = __Token()
class Example:
    __token_t = type(_token)

    def __init__(self,_direct_construction_is_not_allowed) -> None:
        """
        We provide some other class method to construct it.
        Here are them:
            create()
        """
        if type(_direct_construction_is_not_allowed) != self.__token_t:
            raise RuntimeError("Use .create method instead")

    @classmethod
    def create(cls):
        return cls(cls.__token_t()) 
    
_token = None
